### C++ Code

```java
#include <iostream>
using namespace std;

int N;
bool go(int r,int sum);

int main() {
	int T;
	cin >> T;
	int tc;
	for (int tc = 1; tc <= T; tc++) {
		cin >> N;
		int cnt = 0;
		for (int n = 1; n <= N; n++) {
			if (go(n, 0)) {
				cnt++;
			}
		}
		cout << "#" << tc << " " << cnt << endl;
	}
}

bool go(int r, int sum) {
	sum = sum + r;
	if (N < sum) {
		return false;
	}
	else if (N == sum) {
		return true;
	} else {
		return go(r + 1, sum);
	}
}
```
